From Python to Pythonic

Most are lessons learned from pyvideo.

Loop:

# python 2
for a in xrange(5):
    # do sth

# python 3
for a in range(5):
    # do sth

# loop backwards
for a in reversed(range(5)):
    # do sth

# loop over two lists
# in python3 izip is zip
from itertools import izip
list1 = ['a', 'b', 'c']
list2 = [1, 2, 3]
for list1, list2 in izip(list1, list2):
    # do sth
    # do not use zip, memory consuming
# or merge these two lists into dictionary
dict(izip(list1, list2)) # result in {'a':1, ...}

# loop in sorted/revers-sorted
list1 = ['b', 'a', 'c']
for a in sorted(list1):
    # do sth
for a in sorted(list1, reverse = True):
    # do str

# replace while with for
with open(mydir, 'r') as f:
    for line in iter(f.readline(), ''):
        # do sth
        # do not use while True ... 

# remove all flag variables to find a certain variable in for loop
def find(iterable_object, target_object):
    """Find target in iterable object."""
    for item in iterable_object:
        if item == target_object:
            # code like flag = False, if True, flag = True need to be removed
            break
    else: # associated with if but aliend with for
        return False
    return True

# loop through dictionary
for k, v in d.iteritems():
    # do sth
    # do not use iter because make a big list in memory
    # for python3 is d.items()

Dictionary:

# list to dictionary, index as key
list1 = ['a', 'b', 'c']
dict(enumerate(list1)) # result in {0: 'a', ...}

# do not use has key, use get
d = {'a': 1, 'b': 2, 'c': 3}
d.get('d', 0)

# group list of objects to dictionary
lectures = ['machine learning', 'math', 'nature language processing', 'information retrieval']
d = {}
for lecture in lectures:
    key = lecture[0] # group by first lecture
    d.setdefault(key, []).append(lecture)
    # result in {'m':['machine learning', 'math'], ...}

Improve readability:

# bad
my_func('a', 'b', 'c')
# good (a little bit slow)
my_func(arg_name1='a', arg_name2='b', arg_name3='c')

Clarify multiple return values with namedtuple.

from collecitons import namedtuple
# bad
def my_func():
    return ('bo', 11)

# good
def my_func():
    MultipleReturn = collections.namedtuple('ReturnValue', ['name', 'age'])
    return MultipleReturn(name='bo', age=11)

Use deque instead of list for fast get and pop:

# bad, O(n)
characters = ['b', 'c', 'd']
characters.insert(0, 'a')
characters.pop(0)
# good, O(1)
characters = deque(['b', 'c', 'd'])
characters.popleft()
characters.appendleft('a')

Factor out administrate logic from business logic with decorators. For example, you want to build a web crawler (spider), but links in robot.txt are often needed to exclude from your list of urls, so you have something like this:

def crawl_page(url, robot_txt_urls):
    """Crawl current page if it's not belong to robot txt."""
    if url in robot_txt_urls:
        return None
    # otherwise, start to get page content
    page = requests.get(url)
    results =  .... # do crawl
    return results

In this function, the code if ... in … is administrate logic, and the crawling part should be our business logic. A better form should be put administrate logic into a seperate decorator function:

def checker(banned):
    def decorator(func):
        def wrapper(arg):
            if arg in banned:
                return None
            return func(arg)
        return wrapper
    return decorator

banned_lst = ['http://www.google.com/page_not_want_to_be_crawled', 'http://www.google.com/secret']

@checker(banned_lst)
def crawl(url):
    return requests.get(url)

Context Manager, something new:

# bad
try:
    os.remove('i_probably_do_not_exist.txt')
except OSError:
    pass

# good
with ignored(OSError):
    os.remove('i_probably_do_not_exist')

Another example, context manager for pymongo

import pymongo


class Operation(object):
    def __init__(self, uri, db, col):
        """Initialize connection."""
        self._client = pymongo.MongoClient(uri)
        self._col = self._client[db][col]

    def __enter__(self):
        """Get colleciton instance."""
        return self._col

    def __exit__(self, exc_type, exc_val, exc_tb):
        """Close connection."""
        self._client.close()

with Operation(uri=URI, db=DB, col=COL) as col:
    cursor = col.find() # do some query
